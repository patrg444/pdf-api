Easy to Use

import requests

# Generate PDF from HTML
response = requests.post(
    "https://api.lightningpdf.com/api/v1/pdf/generate",
    headers={"X-API-Key": "your_api_key"},
    json={
        "html": "<h1>Hello World</h1>",
        "options": {"format": "A4"}
    }
)

# Download the PDF
pdf_url = response.json()["file_url"]
